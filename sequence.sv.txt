class fifo_sequence extends uvm_sequence #(fifo_item);

    `uvm_object_utils(fifo_sequence)

    function new(string name="fifo_sequence");
        super.new(name);
    endfunction

    virtual task body();
        fifo_item tr;

        repeat (50) begin
            tr = fifo_item::type_id::create("tr");

            assert(tr.randomize());

            start_item(tr);
            finish_item(tr);
        end
    endtask

endclass
